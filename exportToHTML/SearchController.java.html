59       private double calculateAverage(List<Integer> marks) {
60           Integer sum = 0;
61           if(!marks.isEmpty()) {
62               for (Integer mark : marks) {
63                   sum += mark;
64               }
65               return sum.doubleValue() / marks.size();
66           }
